function vowelIndices(word) {
  let arr = ['a', 'e', 'i', 'o', 'u', 'y'];
  let res = [];
  let s = word.toLowerCase();
  for (let i = 0; i < s.length; i++) {

      res.push(i + 1);
    }
  } 
  return res;
}
